import React, { useState, useEffect, useRef } from 'react';
import { Send, Users, MessageCircle, Settings, Plus, X, Upload, Heart, Eye, EyeOff, LogOut, Lock, Shield, Grid3x3, Play, Camera, User, Search, Bell, ChevronRight, Mic, Image, Video, Smile } from 'lucide-react';

const DEFAULT_GROUPS = [
  { id: 'grp-class11a', name: 'Class 11-A 🏫', emoji: '🏫', description: 'Official class group', members: ['everyone'], admins: [], isDefault: true, color: 'from-blue-500 to-cyan-500' },
  { id: 'grp-study', name: 'Study Group 📚', emoji: '📚', description: 'Share notes and doubts', members: ['everyone'], admins: [], isDefault: true, color: 'from-green-500 to-emerald-500' },
  { id: 'grp-sports', name: 'Sports Team ⚽', emoji: '⚽', description: 'Practice schedules & updates', members: ['everyone'], admins: [], isDefault: true, color: 'from-orange-500 to-red-500' },
  { id: 'grp-fun', name: 'Fun & Memes 😂', emoji: '😂', description: 'Timepass group', members: ['everyone'], admins: [], isDefault: true, color: 'from-pink-500 to-purple-500' },
];

export default function SchoolFriendsApp() {
  const [currentUser, setCurrentUser] = useState(null);
  const [showLoginForm, setShowLoginForm] = useState(true);
  const [friends, setFriends] = useState([]);
  const [groups, setGroups] = useState(DEFAULT_GROUPS);
  const [activeSection, setActiveSection] = useState('dms'); // 'dms' or 'groups'
  const [selectedFriend, setSelectedFriend] = useState(null);
  const [selectedGroup, setSelectedGroup] = useState(null);
  const [showNewGroupModal, setShowNewGroupModal] = useState(false);
  const [showProfileModal, setShowProfileModal] = useState(false);
  const [showSettingsModal, setShowSettingsModal] = useState(false);
  const [showViewProfile, setShowViewProfile] = useState(false);
  const [showReelsView, setShowReelsView] = useState(false);
  const [newGroupName, setNewGroupName] = useState('');
  const [messages, setMessages] = useState({});
  const [messageText, setMessageText] = useState('');
  const [privacySetting, setPrivacySetting] = useState('private');
  const [loginEmail, setLoginEmail] = useState('');
  const [loginPassword, setLoginPassword] = useState('');
  const [loginPhone, setLoginPhone] = useState('');
  const [loginName, setLoginName] = useState('');
  const [userProfiles, setUserProfiles] = useState({});
  const [profileTab, setProfileTab] = useState('gallery');
  const [selectedReel, setSelectedReel] = useState(null);
  const [searchQuery, setSearchQuery] = useState('');
  const messagesEndRef = useRef(null);

  useEffect(() => { loadAllData(); }, []);
  useEffect(() => { scrollToBottom(); }, [messages, selectedFriend, selectedGroup]);

  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  };

  const loadAllData = async () => {
    try {
      const usersData = await window.storage?.get('sfapp:users');
      if (usersData?.value) {
        const parsed = JSON.parse(usersData.value);
        if (parsed.length > 0) {
          // Don't auto-login - just show login form
        }
      }

      // Load default group messages
      const grpMsgs = await window.storage?.get('sfapp:groupMessages');
      if (grpMsgs?.value) {
        setMessages(prev => ({ ...prev, ...JSON.parse(grpMsgs.value) }));
      }

      // Load user-created groups
      const userGrps = await window.storage?.get('sfapp:customGroups');
      if (userGrps?.value) {
        setGroups([...DEFAULT_GROUPS, ...JSON.parse(userGrps.value)]);
      }
    } catch (e) {
      console.log('Fresh start');
    }
  };

  const loadUserData = async (user) => {
    try {
      const usersData = await window.storage?.get('sfapp:users');
      if (usersData?.value) {
        const allUsers = JSON.parse(usersData.value);
        const others = allUsers.filter(u => u.id !== user.id);
        setFriends(others);
        
        const profiles = {};
        for (let u of allUsers) {
          const pd = await window.storage?.get(`sfapp:profile:${u.id}`);
          if (pd?.value) profiles[u.id] = JSON.parse(pd.value);
        }
        setUserProfiles(profiles);
      }

      const dmData = await window.storage?.get(`sfapp:dms:${user.id}`);
      if (dmData?.value) {
        setMessages(prev => ({ ...prev, ...JSON.parse(dmData.value) }));
      }
    } catch (e) {}
  };

  const handleLogin = async () => {
    if (!loginEmail || !loginPassword || !loginPhone || !loginName) {
      alert('Saare fields bhar do!');
      return;
    }

    try {
      const usersData = await window.storage?.get('sfapp:users');
      const users = usersData?.value ? JSON.parse(usersData.value) : [];

      // Check if user already exists (login)
      const existing = users.find(u => u.email === loginEmail && u.password === loginPassword);
      if (existing) {
        setCurrentUser(existing);
        setShowLoginForm(false);
        await loadUserData(existing);
        return;
      }

      // New user (register)
      const newUser = {
        id: `user_${Date.now()}`,
        name: loginName,
        email: loginEmail,
        password: loginPassword,
        phone: loginPhone,
        class: '11th',
        joinedDate: new Date().toLocaleDateString(),
      };

      users.push(newUser);
      await window.storage?.set('sfapp:users', JSON.stringify(users));

      const emptyProfile = { bio: 'Hey! 👋', gallery: [], reels: [], profilePic: null };
      await window.storage?.set(`sfapp:profile:${newUser.id}`, JSON.stringify(emptyProfile));

      setCurrentUser(newUser);
      setShowLoginForm(false);
      setLoginEmail(''); setLoginPassword(''); setLoginPhone(''); setLoginName('');
      await loadUserData(newUser);
    } catch (e) {
      alert('Error: ' + e.message);
    }
  };

  const handleAddFriend = async () => {
    const email = prompt("Dost ka email daalo:");
    if (!email) return;

    try {
      const usersData = await window.storage?.get('sfapp:users');
      const users = usersData?.value ? JSON.parse(usersData.value) : [];
      const found = users.find(u => u.email === email);

      if (!found) {
        alert('Yeh user registered nahi hai abhi!');
        return;
      }
      if (found.id === currentUser.id) {
        alert('Apne aap ko friend nahi add kar sakte! 😄');
        return;
      }
      if (friends.find(f => f.id === found.id)) {
        alert('Pehle se friend hai!');
        return;
      }
      setFriends(prev => [...prev, found]);
    } catch (e) {}
  };

  const handleCreateGroup = async () => {
    if (!newGroupName.trim()) return;
    const colors = ['from-violet-500 to-purple-500', 'from-teal-500 to-green-500', 'from-rose-500 to-pink-500', 'from-amber-500 to-orange-500'];
    const newGrp = {
      id: `grp_${Date.now()}`,
      name: newGroupName,
      emoji: '💬',
      description: 'Custom group',
      createdBy: currentUser.id,
      members: [currentUser.id],
      admins: [currentUser.id],
      isDefault: false,
      color: colors[Math.floor(Math.random() * colors.length)],
    };
    const customGroups = groups.filter(g => !g.isDefault);
    customGroups.push(newGrp);
    await window.storage?.set('sfapp:customGroups', JSON.stringify(customGroups));
    setGroups([...DEFAULT_GROUPS, ...customGroups]);
    setNewGroupName('');
    setShowNewGroupModal(false);
  };

  const handleSendMessage = async () => {
    if (!messageText.trim()) return;

    const newMsg = {
      id: `msg_${Date.now()}`,
      sender: currentUser.name,
      senderId: currentUser.id,
      text: messageText,
      timestamp: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }),
      type: 'text'
    };

    const key = selectedFriend
      ? `dm_${[currentUser.id, selectedFriend.id].sort().join('_')}`
      : `grp_${selectedGroup.id}`;

    const existing = messages[key] || [];
    const updated = [...existing, newMsg];

    setMessages(prev => ({ ...prev, [key]: updated }));
    setMessageText('');

    try {
      if (selectedFriend) {
        const dmData = await window.storage?.get(`sfapp:dms:${currentUser.id}`);
        const dms = dmData?.value ? JSON.parse(dmData.value) : {};
        dms[key] = updated;
        await window.storage?.set(`sfapp:dms:${currentUser.id}`, JSON.stringify(dms));
      } else {
        const grpData = await window.storage?.get('sfapp:groupMessages');
        const grpMsgs = grpData?.value ? JSON.parse(grpData.value) : {};
        grpMsgs[key] = updated;
        await window.storage?.set('sfapp:groupMessages', JSON.stringify(grpMsgs));
      }
    } catch (e) {}
  };

  const handleMediaMessage = async (file, chatKey) => {
    if (!file) return;
    const reader = new FileReader();
    reader.onload = async (e) => {
      const isVideo = file.type.startsWith('video/');
      const newMsg = {
        id: `msg_${Date.now()}`,
        sender: currentUser.name,
        senderId: currentUser.id,
        mediaData: e.target.result,
        mediaType: isVideo ? 'video' : 'image',
        timestamp: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }),
        type: 'media'
      };
      const existing = messages[chatKey] || [];
      const updated = [...existing, newMsg];
      setMessages(prev => ({ ...prev, [chatKey]: updated }));
    };
    reader.readAsDataURL(file);
  };

  const handleUploadToProfile = async (file, type) => {
    if (!file) return;
    const reader = new FileReader();
    reader.onload = async (e) => {
      const base64 = e.target.result;
      try {
        const pd = await window.storage?.get(`sfapp:profile:${currentUser.id}`);
        const profile = pd?.value ? JSON.parse(pd.value) : { bio: '', gallery: [], reels: [], profilePic: null };
        const newMedia = { id: Date.now(), data: base64, uploadedDate: new Date().toLocaleDateString(), isVideo: file.type.startsWith('video/') };
        if (type === 'photo') profile.gallery.push(newMedia);
        else if (type === 'reel') profile.reels.push(newMedia);
        else if (type === 'profilepic') profile.profilePic = base64;
        await window.storage?.set(`sfapp:profile:${currentUser.id}`, JSON.stringify(profile));
        setUserProfiles(prev => ({ ...prev, [currentUser.id]: profile }));
        if (type === 'profilepic') setCurrentUser(prev => ({ ...prev, profilePic: base64 }));
      } catch (e) {}
    };
    reader.readAsDataURL(file);
  };

  const handleLogout = () => {
    setCurrentUser(null); setShowLoginForm(true); setFriends([]);
    setGroups(DEFAULT_GROUPS); setMessages({}); setUserProfiles({});
    setSelectedFriend(null); setSelectedGroup(null);
  };

  const myProfile = userProfiles[currentUser?.id] || { bio: '', gallery: [], reels: [], profilePic: null };
  const viewingProfile = selectedFriend ? userProfiles[selectedFriend.id] : null;

  const getCurrentMessages = () => {
    if (selectedFriend) {
      const key = `dm_${[currentUser.id, selectedFriend.id].sort().join('_')}`;
      return { key, msgs: messages[key] || [] };
    }
    if (selectedGroup) {
      const key = `grp_${selectedGroup.id}`;
      return { key, msgs: messages[key] || [] };
    }
    return { key: null, msgs: [] };
  };

  const { key: chatKey, msgs: currentMessages } = currentUser ? getCurrentMessages() : { key: null, msgs: [] };

  const filteredFriends = friends.filter(f =>
    f.name?.toLowerCase().includes(searchQuery.toLowerCase()) ||
    f.email?.toLowerCase().includes(searchQuery.toLowerCase())
  );

  const filteredGroups = groups.filter(g =>
    g.name?.toLowerCase().includes(searchQuery.toLowerCase())
  );

  // ─── LOGIN SCREEN ────────────────────────────────────────────────────────────
  if (showLoginForm) {
    return (
      <div style={{ fontFamily: "'Nunito', sans-serif" }} className="min-h-screen flex items-center justify-center p-4 relative overflow-hidden"
        style={{ background: 'linear-gradient(135deg, #0f0c29, #302b63, #24243e)', fontFamily: "'Nunito', sans-serif" }}>
        <link href="https://fonts.googleapis.com/css2?family=Nunito:wght@400;600;700;800;900&display=swap" rel="stylesheet" />

        {/* Decorative blobs */}
        <div className="absolute top-10 left-10 w-64 h-64 rounded-full opacity-20" style={{ background: 'radial-gradient(circle, #f72585, transparent)' }} />
        <div className="absolute bottom-10 right-10 w-80 h-80 rounded-full opacity-20" style={{ background: 'radial-gradient(circle, #4cc9f0, transparent)' }} />

        <div className="relative z-10 w-full max-w-md">
          {/* Logo */}
          <div className="text-center mb-8">
            <div className="inline-flex items-center justify-center w-20 h-20 rounded-2xl mb-4 shadow-2xl"
              style={{ background: 'linear-gradient(135deg, #f72585, #7209b7)' }}>
              <span className="text-4xl">🏫</span>
            </div>
            <h1 className="text-4xl font-black text-white tracking-tight">FriendsHub</h1>
            <p className="text-purple-300 mt-1">School ka apna social space</p>
          </div>

          <div className="rounded-3xl p-8 shadow-2xl" style={{ background: 'rgba(255,255,255,0.07)', backdropFilter: 'blur(20px)', border: '1px solid rgba(255,255,255,0.1)' }}>
            <div className="space-y-4">
              <input type="text" placeholder="Apna naam" value={loginName}
                onChange={e => setLoginName(e.target.value)}
                className="w-full px-4 py-3 rounded-xl text-white placeholder-purple-300 focus:outline-none focus:ring-2 focus:ring-pink-500"
                style={{ background: 'rgba(255,255,255,0.1)', border: '1px solid rgba(255,255,255,0.15)' }} />
              <input type="email" placeholder="Email address" value={loginEmail}
                onChange={e => setLoginEmail(e.target.value)}
                className="w-full px-4 py-3 rounded-xl text-white placeholder-purple-300 focus:outline-none focus:ring-2 focus:ring-pink-500"
                style={{ background: 'rgba(255,255,255,0.1)', border: '1px solid rgba(255,255,255,0.15)' }} />
              <input type="tel" placeholder="Phone (+91 XXXXX XXXXX)" value={loginPhone}
                onChange={e => setLoginPhone(e.target.value)}
                className="w-full px-4 py-3 rounded-xl text-white placeholder-purple-300 focus:outline-none focus:ring-2 focus:ring-pink-500"
                style={{ background: 'rgba(255,255,255,0.1)', border: '1px solid rgba(255,255,255,0.15)' }} />
              <input type="password" placeholder="Password" value={loginPassword}
                onChange={e => setLoginPassword(e.target.value)}
                onKeyDown={e => e.key === 'Enter' && handleLogin()}
                className="w-full px-4 py-3 rounded-xl text-white placeholder-purple-300 focus:outline-none focus:ring-2 focus:ring-pink-500"
                style={{ background: 'rgba(255,255,255,0.1)', border: '1px solid rgba(255,255,255,0.15)' }} />
              <button onClick={handleLogin}
                className="w-full py-3 rounded-xl font-black text-white text-lg transition-all hover:scale-105 active:scale-95 shadow-lg"
                style={{ background: 'linear-gradient(135deg, #f72585, #7209b7)' }}>
                Login / Register 🚀
              </button>
            </div>
            <p className="text-center text-purple-300 text-sm mt-4">New ho? Register ho jayega automatically!</p>
          </div>
        </div>
      </div>
    );
  }

  // ─── MAIN APP ─────────────────────────────────────────────────────────────────
  return (
    <div className="h-screen flex flex-col overflow-hidden" style={{ fontFamily: "'Nunito', sans-serif", background: '#0f0c29' }}>
      <link href="https://fonts.googleapis.com/css2?family=Nunito:wght@400;600;700;800;900&display=swap" rel="stylesheet" />

      {/* TOP HEADER */}
      <div className="flex items-center justify-between px-4 py-3 flex-shrink-0"
        style={{ background: 'linear-gradient(90deg, #0f0c29, #302b63)', borderBottom: '1px solid rgba(255,255,255,0.1)' }}>
        <div className="flex items-center gap-3">
          <div className="w-10 h-10 rounded-xl flex items-center justify-center text-xl" style={{ background: 'linear-gradient(135deg, #f72585, #7209b7)' }}>🏫</div>
          <div>
            <h1 className="text-white font-black text-lg leading-none">FriendsHub</h1>
            <p className="text-xs" style={{ color: '#a78bfa' }}>Hey, {currentUser?.name}! 👋</p>
          </div>
        </div>
        <div className="flex gap-2">
          <button onClick={() => { setShowProfileModal(true); setShowViewProfile(false); setProfileTab('gallery'); }}
            className="p-2 rounded-xl transition hover:scale-110" style={{ background: 'rgba(255,255,255,0.1)' }}>
            <Camera size={20} className="text-white" />
          </button>
          <button onClick={() => setShowSettingsModal(true)}
            className="p-2 rounded-xl transition hover:scale-110" style={{ background: 'rgba(255,255,255,0.1)' }}>
            <Settings size={20} className="text-white" />
          </button>
          <button onClick={handleLogout}
            className="p-2 rounded-xl transition hover:scale-110" style={{ background: 'rgba(255,255,255,0.1)' }}>
            <LogOut size={20} className="text-white" />
          </button>
        </div>
      </div>

      <div className="flex flex-1 overflow-hidden">
        {/* LEFT SIDEBAR */}
        <div className="w-72 flex flex-col flex-shrink-0" style={{ background: '#1a1730', borderRight: '1px solid rgba(255,255,255,0.08)' }}>

          {/* Search */}
          <div className="p-3">
            <div className="flex items-center gap-2 px-3 py-2 rounded-xl" style={{ background: 'rgba(255,255,255,0.07)' }}>
              <Search size={16} style={{ color: '#a78bfa' }} />
              <input placeholder="Search..." value={searchQuery} onChange={e => setSearchQuery(e.target.value)}
                className="bg-transparent text-white text-sm flex-1 focus:outline-none placeholder-purple-400" />
            </div>
          </div>

          {/* Section Tabs */}
          <div className="flex mx-3 mb-3 rounded-xl overflow-hidden" style={{ background: 'rgba(255,255,255,0.05)' }}>
            <button onClick={() => setActiveSection('dms')}
              className="flex-1 py-2 text-sm font-bold transition"
              style={{ background: activeSection === 'dms' ? 'linear-gradient(135deg, #f72585, #7209b7)' : 'transparent', color: activeSection === 'dms' ? 'white' : '#a78bfa', borderRadius: '10px' }}>
              💬 DMs
            </button>
            <button onClick={() => setActiveSection('groups')}
              className="flex-1 py-2 text-sm font-bold transition"
              style={{ background: activeSection === 'groups' ? 'linear-gradient(135deg, #4361ee, #4cc9f0)' : 'transparent', color: activeSection === 'groups' ? 'white' : '#a78bfa', borderRadius: '10px' }}>
              👥 Groups
            </button>
          </div>

          {/* DMs Section */}
          {activeSection === 'dms' && (
            <div className="flex-1 overflow-y-auto">
              <button onClick={handleAddFriend}
                className="w-full mx-0 px-4 py-2.5 flex items-center gap-2 font-bold text-sm transition hover:opacity-80"
                style={{ background: 'linear-gradient(90deg, rgba(247,37,133,0.2), rgba(114,9,183,0.2))', color: '#f72585' }}>
                <Plus size={18} /> Add Friend
              </button>
              <div className="p-2 space-y-1">
                {filteredFriends.length === 0 && (
                  <p className="text-center py-8 text-sm" style={{ color: '#6b7280' }}>
                    Koi dost nahi abhi 😅<br />
                    <span style={{ color: '#a78bfa' }}>Upar se add karo!</span>
                  </p>
                )}
                {filteredFriends.map(friend => {
                  const dmKey = `dm_${[currentUser.id, friend.id].sort().join('_')}`;
                  const lastMsg = (messages[dmKey] || []).slice(-1)[0];
                  const isSelected = selectedFriend?.id === friend.id && !showViewProfile;
                  return (
                    <button key={friend.id} onClick={() => { setSelectedFriend(friend); setSelectedGroup(null); setShowViewProfile(false); }}
                      className="w-full p-3 rounded-xl flex items-center gap-3 transition"
                      style={{ background: isSelected ? 'rgba(247,37,133,0.2)' : 'transparent', border: isSelected ? '1px solid rgba(247,37,133,0.4)' : '1px solid transparent' }}>
                      <div className="relative flex-shrink-0">
                        {userProfiles[friend.id]?.profilePic
                          ? <img src={userProfiles[friend.id].profilePic} className="w-11 h-11 rounded-full object-cover" />
                          : <div className="w-11 h-11 rounded-full flex items-center justify-center font-black text-white text-lg"
                              style={{ background: 'linear-gradient(135deg, #f72585, #7209b7)' }}>
                              {friend.name?.[0]?.toUpperCase() || friend.email[0].toUpperCase()}
                            </div>}
                        <div className="absolute bottom-0 right-0 w-3 h-3 rounded-full border-2 border-gray-800" style={{ background: '#22c55e' }} />
                      </div>
                      <div className="flex-1 min-w-0 text-left">
                        <p className="font-bold text-sm text-white truncate">{friend.name || friend.email}</p>
                        <p className="text-xs truncate" style={{ color: '#6b7280' }}>{lastMsg ? lastMsg.text : 'Baat shuru karo!'}</p>
                      </div>
                      <button onClick={e => { e.stopPropagation(); setSelectedFriend(friend); setShowViewProfile(true); setShowProfileModal(true); setProfileTab('gallery'); }}
                        className="text-xs px-2 py-1 rounded-lg font-bold" style={{ background: 'rgba(167,139,250,0.2)', color: '#a78bfa' }}>
                        View
                      </button>
                    </button>
                  );
                })}
              </div>
            </div>
          )}

          {/* Groups Section */}
          {activeSection === 'groups' && (
            <div className="flex-1 overflow-y-auto">
              <button onClick={() => setShowNewGroupModal(true)}
                className="w-full px-4 py-2.5 flex items-center gap-2 font-bold text-sm transition hover:opacity-80"
                style={{ background: 'linear-gradient(90deg, rgba(67,97,238,0.2), rgba(76,201,240,0.2))', color: '#4cc9f0' }}>
                <Plus size={18} /> New Group
              </button>
              <div className="p-2 space-y-1">
                <p className="text-xs font-bold px-2 py-1" style={{ color: '#6b7280' }}>DEFAULT GROUPS</p>
                {filteredGroups.filter(g => g.isDefault).map(grp => {
                  const grpKey = `grp_${grp.id}`;
                  const lastMsg = (messages[grpKey] || []).slice(-1)[0];
                  const isSelected = selectedGroup?.id === grp.id;
                  return (
                    <button key={grp.id} onClick={() => { setSelectedGroup(grp); setSelectedFriend(null); }}
                      className="w-full p-3 rounded-xl flex items-center gap-3 transition"
                      style={{ background: isSelected ? 'rgba(67,97,238,0.3)' : 'transparent', border: isSelected ? '1px solid rgba(76,201,240,0.4)' : '1px solid transparent' }}>
                      <div className={`w-11 h-11 rounded-xl flex items-center justify-center text-2xl flex-shrink-0 bg-gradient-to-br ${grp.color}`}>
                        {grp.emoji}
                      </div>
                      <div className="flex-1 min-w-0 text-left">
                        <p className="font-bold text-sm text-white truncate">{grp.name}</p>
                        <p className="text-xs truncate" style={{ color: '#6b7280' }}>{lastMsg ? `${lastMsg.sender}: ${lastMsg.text}` : grp.description}</p>
                      </div>
                    </button>
                  );
                })}

                {filteredGroups.filter(g => !g.isDefault).length > 0 && (
                  <>
                    <p className="text-xs font-bold px-2 py-1 mt-2" style={{ color: '#6b7280' }}>MY GROUPS</p>
                    {filteredGroups.filter(g => !g.isDefault).map(grp => {
                      const grpKey = `grp_${grp.id}`;
                      const lastMsg = (messages[grpKey] || []).slice(-1)[0];
                      const isSelected = selectedGroup?.id === grp.id;
                      return (
                        <button key={grp.id} onClick={() => { setSelectedGroup(grp); setSelectedFriend(null); }}
                          className="w-full p-3 rounded-xl flex items-center gap-3 transition"
                          style={{ background: isSelected ? 'rgba(67,97,238,0.3)' : 'transparent', border: isSelected ? '1px solid rgba(76,201,240,0.4)' : '1px solid transparent' }}>
                          <div className={`w-11 h-11 rounded-xl flex items-center justify-center text-2xl flex-shrink-0 bg-gradient-to-br ${grp.color}`}>
                            {grp.emoji}
                          </div>
                          <div className="flex-1 min-w-0 text-left">
                            <p className="font-bold text-sm text-white truncate">{grp.name}</p>
                            <p className="text-xs truncate" style={{ color: '#6b7280' }}>{lastMsg ? `${lastMsg.sender}: ${lastMsg.text}` : grp.description}</p>
                          </div>
                        </button>
                      );
                    })}
                  </>
                )}
              </div>
            </div>
          )}
        </div>

        {/* MAIN CHAT AREA */}
        <div className="flex-1 flex flex-col overflow-hidden" style={{ background: '#0f0c29' }}>
          {selectedFriend || selectedGroup ? (
            <>
              {/* Chat Header */}
              <div className="px-4 py-3 flex items-center gap-3 flex-shrink-0"
                style={{ background: 'rgba(255,255,255,0.04)', borderBottom: '1px solid rgba(255,255,255,0.08)' }}>
                {selectedFriend ? (
                  <>
                    {userProfiles[selectedFriend.id]?.profilePic
                      ? <img src={userProfiles[selectedFriend.id].profilePic} className="w-10 h-10 rounded-full object-cover" />
                      : <div className="w-10 h-10 rounded-full flex items-center justify-center font-black text-white"
                          style={{ background: 'linear-gradient(135deg, #f72585, #7209b7)' }}>
                          {selectedFriend.name?.[0]?.toUpperCase()}
                        </div>}
                    <div className="flex-1">
                      <p className="font-black text-white">{selectedFriend.name || selectedFriend.email}</p>
                      <p className="text-xs" style={{ color: '#22c55e' }}>● Online</p>
                    </div>
                    <div className="flex items-center gap-1 px-2 py-1 rounded-lg text-xs font-bold"
                      style={{ background: 'rgba(247,37,133,0.15)', color: '#f72585' }}>
                      💬 Personal Chat
                    </div>
                    <button onClick={() => { setShowViewProfile(true); setShowProfileModal(true); setProfileTab('gallery'); }}
                      className="p-2 rounded-xl" style={{ background: 'rgba(255,255,255,0.08)', color: '#a78bfa' }}>
                      <User size={18} />
                    </button>
                  </>
                ) : (
                  <>
                    <div className={`w-10 h-10 rounded-xl flex items-center justify-center text-xl bg-gradient-to-br ${selectedGroup.color}`}>
                      {selectedGroup.emoji}
                    </div>
                    <div className="flex-1">
                      <p className="font-black text-white">{selectedGroup.name}</p>
                      <p className="text-xs" style={{ color: '#a78bfa' }}>{selectedGroup.description}</p>
                    </div>
                    <div className="flex items-center gap-1 px-2 py-1 rounded-lg text-xs font-bold"
                      style={{ background: 'rgba(76,201,240,0.15)', color: '#4cc9f0' }}>
                      👥 Group Chat
                    </div>
                  </>
                )}
              </div>

              {/* Messages */}
              <div className="flex-1 overflow-y-auto p-4 space-y-3">
                {currentMessages.length === 0 && (
                  <div className="flex flex-col items-center justify-center h-full gap-3">
                    <div className="text-6xl">{selectedFriend ? '💬' : selectedGroup.emoji}</div>
                    <p className="font-bold text-white text-lg">
                      {selectedFriend ? `${selectedFriend.name} ke saath chat shuru karo!` : `${selectedGroup.name} mein welcome!`}
                    </p>
                    <p style={{ color: '#6b7280' }}>Pehla message bhejo 👇</p>
                  </div>
                )}
                {currentMessages.map(msg => {
                  const isMine = msg.senderId === currentUser.id;
                  return (
                    <div key={msg.id} className={`flex gap-2 ${isMine ? 'flex-row-reverse' : 'flex-row'}`}>
                      {!isMine && (
                        <div className="w-7 h-7 rounded-full flex items-center justify-center font-black text-white text-xs flex-shrink-0 self-end"
                          style={{ background: 'linear-gradient(135deg, #4361ee, #4cc9f0)' }}>
                          {msg.sender?.[0]?.toUpperCase()}
                        </div>
                      )}
                      <div className={`max-w-xs lg:max-w-md ${isMine ? 'items-end' : 'items-start'} flex flex-col gap-1`}>
                        {!isMine && selectedGroup && (
                          <span className="text-xs font-bold" style={{ color: '#a78bfa' }}>{msg.sender}</span>
                        )}
                        {msg.type === 'media' ? (
                          <div className="rounded-2xl overflow-hidden" style={{ maxWidth: '240px' }}>
                            {msg.mediaType === 'video'
                              ? <video src={msg.mediaData} controls className="w-full rounded-2xl" style={{ maxHeight: '200px' }} />
                              : <img src={msg.mediaData} className="w-full rounded-2xl" style={{ maxHeight: '200px', objectFit: 'cover' }} />}
                          </div>
                        ) : (
                          <div className="px-4 py-2.5 rounded-2xl"
                            style={{
                              background: isMine ? 'linear-gradient(135deg, #f72585, #7209b7)' : 'rgba(255,255,255,0.1)',
                              color: 'white',
                              borderBottomRightRadius: isMine ? '4px' : '16px',
                              borderBottomLeftRadius: isMine ? '16px' : '4px',
                            }}>
                            <p className="text-sm">{msg.text}</p>
                          </div>
                        )}
                        <span className="text-xs px-1" style={{ color: '#4b5563' }}>{msg.timestamp}</span>
                      </div>
                    </div>
                  );
                })}
                <div ref={messagesEndRef} />
              </div>

              {/* Message Input */}
              <div className="p-3 flex-shrink-0" style={{ background: 'rgba(255,255,255,0.04)', borderTop: '1px solid rgba(255,255,255,0.08)' }}>
                <div className="flex items-center gap-2">
                  {/* Media Upload */}
                  <label className="cursor-pointer p-2 rounded-xl transition hover:scale-110" style={{ background: 'rgba(255,255,255,0.08)' }}>
                    <input type="file" accept="image/*,video/*" className="hidden"
                      onChange={e => handleMediaMessage(e.target.files?.[0], chatKey)} />
                    <Image size={20} style={{ color: '#a78bfa' }} />
                  </label>

                  <input type="text" value={messageText} onChange={e => setMessageText(e.target.value)}
                    onKeyDown={e => e.key === 'Enter' && handleSendMessage()}
                    placeholder={`${selectedFriend ? selectedFriend.name : selectedGroup?.name} ko message karo...`}
                    className="flex-1 px-4 py-2.5 rounded-xl text-white text-sm focus:outline-none"
                    style={{ background: 'rgba(255,255,255,0.08)', border: '1px solid rgba(255,255,255,0.1)', color: 'white' }} />

                  <button onClick={handleSendMessage}
                    className="p-2.5 rounded-xl transition hover:scale-110 active:scale-95"
                    style={{ background: messageText.trim() ? 'linear-gradient(135deg, #f72585, #7209b7)' : 'rgba(255,255,255,0.08)' }}>
                    <Send size={20} className="text-white" />
                  </button>
                </div>
              </div>
            </>
          ) : (
            <div className="flex-1 flex flex-col items-center justify-center gap-4">
              <div className="text-8xl">🏫</div>
              <h2 className="text-2xl font-black text-white">FriendsHub</h2>
              <p style={{ color: '#6b7280' }}>Koi friend ya group select karo baat karne ke liye</p>
              <div className="flex gap-3 mt-2">
                <button onClick={() => setActiveSection('dms')}
                  className="px-4 py-2 rounded-xl font-bold text-sm" style={{ background: 'rgba(247,37,133,0.2)', color: '#f72585' }}>
                  💬 Friends
                </button>
                <button onClick={() => setActiveSection('groups')}
                  className="px-4 py-2 rounded-xl font-bold text-sm" style={{ background: 'rgba(76,201,240,0.2)', color: '#4cc9f0' }}>
                  👥 Groups
                </button>
              </div>
            </div>
          )}
        </div>
      </div>

      {/* ── NEW GROUP MODAL ── */}
      {showNewGroupModal && (
        <div className="fixed inset-0 flex items-center justify-center p-4 z-50" style={{ background: 'rgba(0,0,0,0.7)' }}>
          <div className="w-full max-w-sm rounded-2xl p-6" style={{ background: '#1a1730', border: '1px solid rgba(255,255,255,0.1)' }}>
            <div className="flex justify-between items-center mb-4">
              <h3 className="text-xl font-black text-white">New Group</h3>
              <button onClick={() => setShowNewGroupModal(false)}><X size={22} style={{ color: '#a78bfa' }} /></button>
            </div>
            <input value={newGroupName} onChange={e => setNewGroupName(e.target.value)}
              placeholder="Group ka naam daalo..."
              className="w-full px-4 py-3 rounded-xl mb-4 text-white focus:outline-none"
              style={{ background: 'rgba(255,255,255,0.08)', border: '1px solid rgba(255,255,255,0.1)' }} />
            <div className="flex gap-2">
              <button onClick={handleCreateGroup}
                className="flex-1 py-3 rounded-xl font-black text-white"
                style={{ background: 'linear-gradient(135deg, #4361ee, #4cc9f0)' }}>
                Create ✨
              </button>
              <button onClick={() => setShowNewGroupModal(false)}
                className="flex-1 py-3 rounded-xl font-bold"
                style={{ background: 'rgba(255,255,255,0.08)', color: '#a78bfa' }}>
                Cancel
              </button>
            </div>
          </div>
        </div>
      )}

      {/* ── PROFILE MODAL ── */}
      {showProfileModal && (
        <div className="fixed inset-0 flex items-center justify-center p-4 z-50 overflow-y-auto" style={{ background: 'rgba(0,0,0,0.8)' }}>
          <div className="w-full max-w-lg rounded-3xl overflow-hidden my-4" style={{ background: '#1a1730', border: '1px solid rgba(255,255,255,0.1)' }}>
            {/* Cover */}
            <div className="h-28 relative" style={{ background: 'linear-gradient(135deg, #f72585, #7209b7, #4361ee)' }}>
              <button onClick={() => setShowProfileModal(false)}
                className="absolute top-3 right-3 w-8 h-8 rounded-full flex items-center justify-center"
                style={{ background: 'rgba(0,0,0,0.5)' }}>
                <X size={18} className="text-white" />
              </button>
            </div>

            <div className="px-6 pb-6">
              <div className="flex justify-between items-end -mt-12 mb-4">
                <div>
                  {(showViewProfile ? viewingProfile?.profilePic : myProfile.profilePic)
                    ? <img src={showViewProfile ? viewingProfile.profilePic : myProfile.profilePic}
                        className="w-20 h-20 rounded-2xl object-cover border-4" style={{ borderColor: '#1a1730' }} />
                    : <div className="w-20 h-20 rounded-2xl flex items-center justify-center text-3xl font-black text-white border-4"
                        style={{ background: 'linear-gradient(135deg, #f72585, #7209b7)', borderColor: '#1a1730' }}>
                        {(showViewProfile ? selectedFriend?.name : currentUser?.name)?.[0]?.toUpperCase()}
                      </div>}
                </div>
                {!showViewProfile && (
                  <label className="cursor-pointer">
                    <input type="file" accept="image/*" className="hidden"
                      onChange={e => handleUploadToProfile(e.target.files?.[0], 'profilepic')} />
                    <div className="px-3 py-1.5 rounded-xl text-xs font-bold cursor-pointer"
                      style={{ background: 'rgba(247,37,133,0.2)', color: '#f72585' }}>
                      📷 Change Pic
                    </div>
                  </label>
                )}
              </div>

              <h2 className="text-xl font-black text-white">
                {showViewProfile ? (selectedFriend?.name || selectedFriend?.email) : currentUser?.name}
              </h2>
              <p className="text-sm mb-4" style={{ color: '#a78bfa' }}>
                {showViewProfile ? viewingProfile?.bio : myProfile.bio || 'Hey! 👋'}
              </p>

              <div className="flex gap-6 mb-6 p-3 rounded-xl" style={{ background: 'rgba(255,255,255,0.05)' }}>
                {[
                  { label: 'Photos', val: (showViewProfile ? viewingProfile : myProfile)?.gallery?.length || 0 },
                  { label: 'Reels', val: (showViewProfile ? viewingProfile : myProfile)?.reels?.length || 0 },
                  { label: 'Joined', val: (showViewProfile ? selectedFriend : currentUser)?.joinedDate || '—' },
                ].map(s => (
                  <div key={s.label} className="text-center flex-1">
                    <p className="font-black text-white text-lg">{s.val}</p>
                    <p className="text-xs" style={{ color: '#6b7280' }}>{s.label}</p>
                  </div>
                ))}
              </div>

              {/* Upload buttons (my profile only) */}
              {!showViewProfile && (
                <div className="grid grid-cols-2 gap-2 mb-4">
                  <label className="cursor-pointer">
                    <input type="file" accept="image/*" className="hidden"
                      onChange={e => handleUploadToProfile(e.target.files?.[0], 'photo')} />
                    <div className="py-2.5 rounded-xl text-center text-sm font-bold cursor-pointer"
                      style={{ background: 'rgba(76,201,240,0.15)', color: '#4cc9f0' }}>
                      🖼️ Add Photo
                    </div>
                  </label>
                  <label className="cursor-pointer">
                    <input type="file" accept="image/*,video/*" className="hidden"
                      onChange={e => handleUploadToProfile(e.target.files?.[0], 'reel')} />
                    <div className="py-2.5 rounded-xl text-center text-sm font-bold cursor-pointer"
                      style={{ background: 'rgba(247,37,133,0.15)', color: '#f72585' }}>
                      🎬 Add Reel
                    </div>
                  </label>
                </div>
              )}

              {/* Tabs */}
              <div className="flex rounded-xl overflow-hidden mb-4" style={{ background: 'rgba(255,255,255,0.05)' }}>
                {['gallery', 'reels'].map(t => (
                  <button key={t} onClick={() => setProfileTab(t)}
                    className="flex-1 py-2 text-sm font-bold capitalize"
                    style={{ background: profileTab === t ? 'linear-gradient(135deg, #f72585, #7209b7)' : 'transparent', color: profileTab === t ? 'white' : '#6b7280' }}>
                    {t === 'gallery' ? '🖼️ Gallery' : '🎬 Reels'}
                  </button>
                ))}
              </div>

              {/* Gallery */}
              {profileTab === 'gallery' && (
                <div>
                  {((showViewProfile ? viewingProfile : myProfile)?.gallery?.length === 0) ? (
                    <p className="text-center py-8 text-sm" style={{ color: '#6b7280' }}>Koi photos nahi abhi</p>
                  ) : (
                    <div className="grid grid-cols-3 gap-2">
                      {(showViewProfile ? viewingProfile : myProfile)?.gallery?.map(photo => (
                        <div key={photo.id} className="aspect-square rounded-xl overflow-hidden">
                          <img src={photo.data} className="w-full h-full object-cover" />
                        </div>
                      ))}
                    </div>
                  )}
                </div>
              )}

              {/* Reels */}
              {profileTab === 'reels' && (
                <div>
                  {((showViewProfile ? viewingProfile : myProfile)?.reels?.length === 0) ? (
                    <p className="text-center py-8 text-sm" style={{ color: '#6b7280' }}>Koi reels nahi abhi</p>
                  ) : (
                    <div className="grid grid-cols-2 gap-3">
                      {(showViewProfile ? viewingProfile : myProfile)?.reels?.map(reel => (
                        <div key={reel.id} onClick={() => { setSelectedReel(reel); setShowReelsView(true); }}
                          className="aspect-video rounded-xl overflow-hidden relative cursor-pointer group"
                          style={{ background: '#000' }}>
                          {reel.isVideo
                            ? <video src={reel.data} className="w-full h-full object-cover opacity-70" />
                            : <img src={reel.data} className="w-full h-full object-cover" />}
                          <div className="absolute inset-0 flex items-center justify-center">
                            <div className="w-12 h-12 rounded-full flex items-center justify-center" style={{ background: 'rgba(0,0,0,0.6)' }}>
                              <Play size={24} className="text-white fill-white" />
                            </div>
                          </div>
                        </div>
                      ))}
                    </div>
                  )}
                </div>
              )}
            </div>
          </div>
        </div>
      )}

      {/* ── REEL VIEWER ── */}
      {showReelsView && selectedReel && (
        <div className="fixed inset-0 flex items-center justify-center p-4 z-[60]" style={{ background: 'rgba(0,0,0,0.95)' }}>
          <button onClick={() => { setShowReelsView(false); setSelectedReel(null); }}
            className="absolute top-4 right-4 w-10 h-10 rounded-full flex items-center justify-center z-10"
            style={{ background: 'rgba(255,255,255,0.2)' }}>
            <X size={22} className="text-white" />
          </button>
          <div className="w-full max-w-sm rounded-2xl overflow-hidden">
            {selectedReel.isVideo
              ? <video src={selectedReel.data} controls autoPlay className="w-full" style={{ maxHeight: '80vh' }} />
              : <img src={selectedReel.data} className="w-full rounded-2xl" style={{ maxHeight: '80vh', objectFit: 'contain' }} />}
            <p className="text-center text-sm mt-3" style={{ color: '#6b7280' }}>Uploaded: {selectedReel.uploadedDate}</p>
          </div>
        </div>
      )}

      {/* ── SETTINGS MODAL ── */}
      {showSettingsModal && (
        <div className="fixed inset-0 flex items-center justify-center p-4 z-50" style={{ background: 'rgba(0,0,0,0.7)' }}>
          <div className="w-full max-w-sm rounded-2xl p-6" style={{ background: '#1a1730', border: '1px solid rgba(255,255,255,0.1)' }}>
            <div className="flex justify-between items-center mb-6">
              <h3 className="text-xl font-black text-white">Settings ⚙️</h3>
              <button onClick={() => setShowSettingsModal(false)}><X size={22} style={{ color: '#a78bfa' }} /></button>
            </div>
            <div className="space-y-3 mb-6">
              <p className="text-sm font-bold" style={{ color: '#a78bfa' }}>Friends List Privacy</p>
              {[
                { val: 'private', label: '🔒 Private', desc: 'Sirf tum dekh sakte ho' },
                { val: 'shared', label: '👁️ Public', desc: 'Sabko dikhega' },
              ].map(opt => (
                <label key={opt.val} className="flex items-center gap-3 p-3 rounded-xl cursor-pointer"
                  style={{ background: privacySetting === opt.val ? 'rgba(247,37,133,0.15)' : 'rgba(255,255,255,0.05)', border: `1px solid ${privacySetting === opt.val ? 'rgba(247,37,133,0.4)' : 'transparent'}` }}>
                  <input type="radio" name="privacy" value={opt.val} checked={privacySetting === opt.val}
                    onChange={e => setPrivacySetting(e.target.value)} className="accent-pink-500" />
                  <div>
                    <p className="text-white text-sm font-bold">{opt.label}</p>
                    <p className="text-xs" style={{ color: '#6b7280' }}>{opt.desc}</p>
                  </div>
                </label>
              ))}
            </div>
            <button onClick={async () => {
              try {
                await window.storage?.set(`sfapp:privacy:${currentUser.id}`, privacySetting);
                alert('Saved! ✅');
                setShowSettingsModal(false);
              } catch (e) {}
            }}
              className="w-full py-3 rounded-xl font-black text-white"
              style={{ background: 'linear-gradient(135deg, #f72585, #7209b7)' }}>
              Save Settings
            </button>
          </div>
        </div>
      )}
    </div>
  );
}
